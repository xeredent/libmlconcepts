    def test_load_wrong_types(self): # noqa: D102
        X = numpy.array(
            [
                [2.0, 3.25, 2.4],
                [9.5, 1.33, 1.22]
            ],
            order = "C",
            dtype = numpy.float32
        )
        Xc = numpy.array(
            [ [1], [0] ],
            order = "C",
            dtype = numpy.int8
        )
        d = numpy_load(X, Xc = Xc)
        assert X.flags.f_contiguous
        assert X.dtype == numpy.float64
        assert Xc.flags.f_contiguous
        assert Xc.dtype == numpy.int32
        assert X.shape == (2, 3)
        assert Xc.shape == (2, 1)
